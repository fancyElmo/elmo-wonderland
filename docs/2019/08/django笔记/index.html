Django入门笔记

2019-08-03 入门知识 django 0 评论 字数统计:949字 阅读时长:2分

修改配置文件

  • 避免使用关键子来，创建py文件、文件夹，等等(admin, django,this, that, test)

  • 注册子应用

    INSTALLED_APPS = [
    'goods'  # 子应用名字
]

  • 注释掉CSRF

    MIDDLEWARE = [
    'django.middleware.security.SecurityMiddleware',
    'django.contrib.sessions.middleware.SessionMiddleware',
    'django.middleware.common.CommonMiddleware',
    # 'django.middleware.csrf.CsrfViewMiddleware',
    'django.contrib.auth.middleware.AuthenticationMiddleware',
    'django.contrib.messages.middleware.MessageMiddleware',
    'django.middleware.clickjacking.XFrameOptionsMiddleware',
]

  • 模板文件夹配置

    TEMPLATES = [
    {
        'BACKEND': 'django.template.backends.django.DjangoTemplates',
        # 模板文件夹配置
        'DIRS': [os.path.join(BASE_DIR, 'templates')],
        'APP_DIRS': True,
        'OPTIONS': {
            'context_processors': [
                'django.template.context_processors.debug',
                'django.template.context_processors.request',
                'django.contrib.auth.context_processors.auth',
                'django.contrib.messages.context_processors.messages',
            ],
        },
    },
]

  • 配置mysql数据库

    DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.mysql',
        'HOST': '127.0.0.1',
        'PORT': 3306,
        'USER': 'root',
        'PASSWORD': '12345678',
        'NAME': 'fuxi1903'
    }
}

  • 修改时区、语言

    LANGUAGE_CODE = 'zh-Hans'

TIME_ZONE = 'Asia/Shanghai'

USE_TZ = False

  • 静态文件夹

    STATICFILES_DIRS = [os.path.join(BASE_DIR, 'static')]

  • 主应用的__init__.py中添加

    import pymysql
pymysql.install_as_MySQLdb()

重定向

from django.shortcuts import render,HttpResponse, redirect, reverse
from django.views import View
from django.http import JsonResponse,HttpResponse


# Create your views here.
class IndexView(View):
    def get(self, reqeust):
        name = 'ZhangSan'

        return render(reqeust, 'index.html',context={'name': name})


class LoginView(View):
    def get(self, request):
        # 登录操作，登录成功后，跳转到首页
        print('登录操作')

        # 跳转方式：1
        # return redirect('http://127.0.0.1:8000/goods/index/')

        # 跳转方式：2
        # reverse功能等同于  flask:url_for
        # return redirect(reverse('goods:index'))

        # 跳转方式：3 带参数的跳转
        # return redirect('/goods/detail/123')

        # 跳转方式：4 带参数的跳转
        return redirect(reverse('goods:detail', kwargs={'id': 456}))


class GoodsDetailView(View):
    def get(self, request, id):
        print('ID 为 %d' % id)
        return render(request, 'detail.html')

模板中的反向解析

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>登录页</title>
</head>
<body>
{# 方式1： 写入url路径 #}
<a href="/goods/detail/1314">跳转到detail页</a>
{# 方式2： 反向解析（带参数） #}
<br>
<a href="{% url 'goods:detail' 520 %}">反向解析跳转（有参数）</a>
<br>
{# 方式3： 反向解析 （无参数） #}
<a href="{% url 'goods:index' %}">反向无参数</a>
</body>
</html>

Cookie

  • 设置cookie
response.set_cookie("username","jack",expires=expires)
# 设置过期时间为100秒
res.set_cookie('user_id', 7788250, expires=100)
  • 获取
	#定义变量获取cookies属性
    cookies = request.COOKIES
    #从cookies通过key获取value
    username = cookies.get("username","未拿到用户名")
  • 删除
	#定义response对象
    response = HttpResponse("删除cookie成功")
    #调用delete_cookie()方法来删除cookie
    response.delete_cookie("username")

Session

  • 设置
# 设置session
request.session["name"] = "root"

# 设置session过期时间为60秒
request.session.set_expiry(10)
  • 获取
# 获取session
name = request.sessin["name"]

name = request.session.get("name")
  • 删除
 request.session.pop('user_id')

过滤器

  • safe，禁用转义，告诉模板这个变量是安全的，可以解释执行
  • length，长度，返回字符串包含字符的个数，或列表、元组、字典的元素个数。
  • default，默认值，如果变量不存在时则返回默认值。
