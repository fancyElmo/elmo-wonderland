Django 分页器

初始化操作

class Paginator(object):
    """

    object_list:可以是列表，元组，查询集或其他含有 count() 或 len()方法的可切片对	象。对于连续的分页，查询集应该有序，例如有order_by()项或默认ordering参数

    per_page:每一页中包含条目数目的最大值，不包括独立成页的那页

    orphans=0:当你使用此参数时说明你不希望最后一页只有很少的条目。如果最后一页的条目数	  少于等于orphans的值，则这些条目会被归并到上一页中（此时的上一页变为最后一页）

    allow_empty_first_page=True： 默认允许第一页为空

    """
    def __init__(self, object_list, per_page, orphans=0, allow_empty_first_page=True)

自定义分页器

----views.py----

from django.core.paginator import Paginator
from models import Book

def mypaginator(request, dataset):
    # dataset = Book.objects.all()  # 数据集

	paginate = Paginator(dataset, 5)  # 实例化Paginator, 每页5条数据

    web_page = request.GET.get('page', 1)  # 获取前端访问

    current_data = paginate.get_page(web_page)  # 得到当前页数据

    # get_page()方法可以检查page是否合法，正确后返回到page()方法

	total_page = paginate.page_range  # 得到全部页数的一个列表

    return current_data, total_page
