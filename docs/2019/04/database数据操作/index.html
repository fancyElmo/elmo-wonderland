MySQL数据操作

2019-04-07 入门知识 mysql 0 评论 字数统计:513字 阅读时长:2分

添加数据

单组添加

-- 全列插入，按顺序在values中填入字段对应数据
-- 注意数据类型要正确
insert into 表名 values();
-- 如果存在自增数据，或者默认值数据，可在对应字段处填写 0、default、null 来简化数据的填写

多组添加

insert into 表名 values(第一组),(第二组),(第三组);

部分值添加

-- 只有允许为空的值可以不写，不允许为空的值必须添加
insert into 表名(字段名1,...) values(值1,...);

修改数据

修改全部数据

update 表名 set 要修改的字段名 = 要修改的数据;

带筛选的数据修改

update 表名 set 要修改的字段名 = 要修改的数据 where 筛选语句;

多个值的修改

update 表名 set 要修改的字段名1 = 要修改的数据1,... where 筛选语句;

数据的查询

查询表的全部数据

select * from 表名;

查询表的部分数据

select 字段名1,... from 表名;

改变字段名的查询

-- as 可以为字段名取别名，表在显示时以别名显示
select 字段名1 as 别名,字段名2 ... from 表名;

带筛选语句的查询

select 字段名1,... from 表名 where 筛选语句;

数据的删除

删除特定数据 - 物理删除

delete from 表名 where 筛选语句;

删除特定数据 - 逻辑删除

-- 先为表添加逻辑删除标记字段
alter table 表名 add is_delete bit default 0;

查看添加后的表结构

删除要逻辑删除的数据

-- 实际上就是为is_delete字段更新数据
update 表名 set is_delete = 1 where 筛选语句;
